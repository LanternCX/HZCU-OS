### 步骤 9：回答问题

1. 理解共享内存和信号量相关的系统调用。  
   这组程序里，共享内存负责“把同一块数据摆在大家都能访问的地方”，信号量负责“决定现在轮到谁动这块数据”。前者解决的是“怎么共享”，后者解决的是“什么时候能读、什么时候能写”。
2. 用 `ipcs` 观察和理解共享内存和信号量信息。  
   运行期间可以在 `ipcs -m` 里看到共享内存，在 `ipcs -s` 里看到信号量，说明这两个对象确实是系统里单独存在的资源。等 `writer` 退出、`reader` 再结束后，对应资源会按程序安排被清掉。
3. 尝试运行多个 `writer`，能否正确同步？请分析代码解释你的判断。  
   这份代码在“多个 `writer` 轮流写同一块共享内存”这个前提下，基本可以正确同步。原因是每个 `writer` 真正写入前都要先等同一个读信号，只有前一条消息被 `reader` 读走后，下一个 `writer` 才能继续写，所以不会同时改写同一块内容。它的限制是一次只能保存一条消息，也没有排队缓冲；如果想同时处理很多写入者，只靠这一块共享内存和这一对读写开关还不够。

## 任务 4：把共享内存实验中的忙等待改成用信号量同步

### 步骤 1：修改文件

把 `"/Users/caoxin/Library/CloudStorage/OneDrive-Personal/Document/HZCU-OS/Lab08/shm/shm3pro.c"` 的全部内容替换成下面这份：

```c
#include <unistd.h>
#include <stdlib.h>
#include <stdio.h>
#include <string.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <sys/sem.h>

#define TEXT_SZ 2048
#define SEM_READ 0
#define SEM_WRITE 1

struct shared_use_st {
    int written_by_you;
    char some_text[TEXT_SZ];
};

static int sem_id = -1;
static int shmid = -1;

static void semaphore_p(int sem_num)
{
    struct sembuf sem_b;

    sem_b.sem_num = sem_num;
    sem_b.sem_op = -1;
    sem_b.sem_flg = SEM_UNDO;

    if (semop(sem_id, &sem_b, 1) == -1) {
        fprintf(stderr, "semop P failed\n");
        exit(EXIT_FAILURE);
    }
}

static void semaphore_v(int sem_num)
{
    struct sembuf sem_b;

    sem_b.sem_num = sem_num;
    sem_b.sem_op = 1;
    sem_b.sem_flg = SEM_UNDO;

    if (semop(sem_id, &sem_b, 1) == -1) {
        fprintf(stderr, "semop V failed\n");
        exit(EXIT_FAILURE);
    }
}

int main()
{
    int running = 1;
    void *shared_memory = (void *)0;
    struct shared_use_st *shared_stuff;
    char buffer[BUFSIZ];

    shmid = shmget((key_t)1234, sizeof(struct shared_use_st), 0666 | IPC_CREAT);
    if (shmid == -1) {
        fprintf(stderr, "shmget failed\n");
        exit(EXIT_FAILURE);
    }

    sem_id = semget((key_t)5678, 2, 0666 | IPC_CREAT);
    if (sem_id == -1) {
        fprintf(stderr, "semget failed\n");
        exit(EXIT_FAILURE);
    }

    shared_memory = shmat(shmid, (void *)0, 0);
    if (shared_memory == (void *)-1) {
        fprintf(stderr, "shmat failed\n");
        exit(EXIT_FAILURE);
    }

    shared_stuff = (struct shared_use_st *)shared_memory;

    while (running) {
        semaphore_p(SEM_WRITE);

        printf("Enter some text: ");
        fflush(stdout);
        fgets(buffer, BUFSIZ, stdin);

        strncpy(shared_stuff->some_text, buffer, TEXT_SZ);
        shared_stuff->some_text[TEXT_SZ - 1] = '\0';
        shared_stuff->written_by_you = 1;

        semaphore_v(SEM_READ);

        if (strncmp(buffer, "end", 3) == 0) {
            running = 0;
        }
    }

    if (shmdt(shared_memory) == -1) {
        fprintf(stderr, "shmdt failed\n");
        exit(EXIT_FAILURE);
    }

    exit(EXIT_SUCCESS);
}
```

### 步骤 2：修改文件

把 `"/Users/caoxin/Library/CloudStorage/OneDrive-Personal/Document/HZCU-OS/Lab08/shm/shm3con.c"` 的全部内容替换成下面这份：

```c
#include <unistd.h>
#include <stdlib.h>
#include <stdio.h>
#include <string.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <sys/sem.h>

#define TEXT_SZ 2048
#define SEM_READ 0
#define SEM_WRITE 1

union semun {
    int val;
    struct semid_ds *buf;
    unsigned short *array;
    struct seminfo *__buf;
};

struct shared_use_st {
    int written_by_you;
    char some_text[TEXT_SZ];
};

static int sem_id = -1;
static int shmid = -1;

static void semaphore_p(int sem_num)
{
    struct sembuf sem_b;

    sem_b.sem_num = sem_num;
    sem_b.sem_op = -1;
    sem_b.sem_flg = SEM_UNDO;

    if (semop(sem_id, &sem_b, 1) == -1) {
        fprintf(stderr, "semop P failed\n");
        exit(EXIT_FAILURE);
    }
}

static void semaphore_v(int sem_num)
{
    struct sembuf sem_b;

    sem_b.sem_num = sem_num;
    sem_b.sem_op = 1;
    sem_b.sem_flg = SEM_UNDO;

    if (semop(sem_id, &sem_b, 1) == -1) {
        fprintf(stderr, "semop V failed\n");
        exit(EXIT_FAILURE);
    }
}

int main()
{
    int running = 1;
    void *shared_memory = (void *)0;
    struct shared_use_st *shared_stuff;
    union semun sem_union;

    shmid = shmget((key_t)1234, sizeof(struct shared_use_st), 0666 | IPC_CREAT);
    if (shmid == -1) {
        fprintf(stderr, "shmget failed\n");
        exit(EXIT_FAILURE);
    }

    sem_id = semget((key_t)5678, 2, 0666 | IPC_CREAT);
    if (sem_id == -1) {
        fprintf(stderr, "semget failed\n");
        exit(EXIT_FAILURE);
    }

    sem_union.val = 0;
    if (semctl(sem_id, SEM_READ, SETVAL, sem_union) == -1) {
        fprintf(stderr, "semctl SETVAL failed\n");
        exit(EXIT_FAILURE);
    }

    sem_union.val = 1;
    if (semctl(sem_id, SEM_WRITE, SETVAL, sem_union) == -1) {
        fprintf(stderr, "semctl SETVAL failed\n");
        exit(EXIT_FAILURE);
    }

    shared_memory = shmat(shmid, (void *)0, 0);
    if (shared_memory == (void *)-1) {
        fprintf(stderr, "shmat failed\n");
        exit(EXIT_FAILURE);
    }

    shared_stuff = (struct shared_use_st *)shared_memory;
    shared_stuff->written_by_you = 0;

    while (running) {
        semaphore_p(SEM_READ);

        if (shared_stuff->written_by_you) {
            printf("You wrote: %s", shared_stuff->some_text);
            shared_stuff->written_by_you = 0;

            if (strncmp(shared_stuff->some_text, "end", 3) == 0) {
                running = 0;
            }
        }

        semaphore_v(SEM_WRITE);
    }

    if (shmdt(shared_memory) == -1) {
        fprintf(stderr, "shmdt failed\n");
        exit(EXIT_FAILURE);
    }

    if (shmctl(shmid, IPC_RMID, 0) == -1) {
        fprintf(stderr, "shmctl(IPC_RMID) failed\n");
        exit(EXIT_FAILURE);
    }

    if (semctl(sem_id, 0, IPC_RMID) == -1) {
        fprintf(stderr, "semctl(IPC_RMID) failed\n");
        exit(EXIT_FAILURE);
    }

    exit(EXIT_SUCCESS);
}
```
