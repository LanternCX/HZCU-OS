### 步骤 4：修改文件

把 `"/Users/caoxin/Library/CloudStorage/OneDrive-Personal/Document/HZCU-OS/Lab08/shm/shm2w.c"` 的全部内容替换成下面这份：

```c
/*write.c*/
#include <stdio.h>
#include <sys/shm.h>
#include <sys/stat.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <stdlib.h>
#include <unistd.h>

#define SHMSZ     27

int main()
{
    char c;
    int shmid;
    key_t key;
    char *shm, *s;

    key = 5679;

    if ((shmid = shmget(key, SHMSZ, IPC_CREAT | 0666)) < 0) {
        perror("shmget");
        exit(1);
    }

    if ((shm = shmat(shmid, NULL, 0)) == (char *) -1) {
        perror("shmat");
        exit(1);
    }

    s = shm;
    for (c = 'a'; c <= 'z'; c++) {
        *s++ = c;
        sleep(1);
    }
    *s = '\0';

    c = getchar();

    return 0;
}
```

这样写入端会每秒只写一个字母，读出端如果在写入过程中启动，就更容易看到共享内存没有自动同步带来的影响。观察完成后，可以再把 `shm2w.c` 恢复成原始版本。

### 步骤 7：回答问题

1. 两个并发进程的启动顺序对程序运行有没有影响？  
   有影响，读出端要等写入端先把共享内存创建出来；如果读出端先跑，通常会因为找不到共享内存而失败。正确顺序是先启动写入端，再启动读出端。
2. 将①替换成 `{*s++ = c;sleep(1);}`，降低写入进程写数据的速度，观察对读出进程的影响？这说明共享内存没有提供同步机制，可能导致数据丢失。  
   写得慢以后，读出端更容易看到不完整或变化中的内容，因为共享内存只负责大家看同一块内存，不负责帮它们排队。也就是说，共享内存本身很快，但不自带“谁先谁后”的保护。

```c
/* Our first program is a consumer. After the headers the shared memory segment
 (the size of our shared memory structure) is created with a call to shmget,
 with the IPC_CREAT bit specified. */

#include <unistd.h>
#include <stdlib.h>
#include <stdio.h>
#include <string.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/shm.h>

#define TEXT_SZ 2048

struct shared_use_st {
    int written_by_you;
    char some_text[TEXT_SZ];
};

int main()
{
    int running = 1;
    void *shared_memory = (void *)0;
    struct shared_use_st *shared_stuff;
    int shmid;

    srand((unsigned int)getpid());

    shmid = shmget((key_t)1234, sizeof(struct shared_use_st), 0666 | IPC_CREAT);

    if (shmid == -1) {
        fprintf(stderr, "shmget failed\n");
        exit(EXIT_FAILURE);
    }

    shared_memory = shmat(shmid, (void *)0, 0);
    if (shared_memory == (void *)-1) {
        fprintf(stderr, "shmat failed\n");
        exit(EXIT_FAILURE);
    }

    shared_stuff = (struct shared_use_st *)shared_memory;
    shared_stuff->written_by_you = 0;
    while (running) {
        if (shared_stuff->written_by_you) {
            printf("You wrote: %s", shared_stuff->some_text);
            shared_stuff->written_by_you = 0;
            if (strncmp(shared_stuff->some_text, "end", 3) == 0) {
                running = 0;
            }
        }
    }

    if (shmdt(shared_memory) == -1) {
        fprintf(stderr, "shmdt failed\n");
        exit(EXIT_FAILURE);
    }

    if (shmctl(shmid, IPC_RMID, 0) == -1) {
        fprintf(stderr, "shmctl(IPC_RMID) failed\n");
        exit(EXIT_FAILURE);
    }

    exit(EXIT_SUCCESS);
}
```

这份代码只去掉了原来消费者打印内容后随机等待的那一行。这样消费者会更快把状态改回“可以继续写入”，便于和原始版本对比。

## 任务 4：编写父进程和子进程通过共享内存交换信息的程序

### 步骤 1：修改文件

在 `"/Users/caoxin/Library/CloudStorage/OneDrive-Personal/Document/HZCU-OS/Lab08/shm/"` 下新建 `shm4.c`，内容如下：

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <sys/wait.h>

struct shared_data {
    int state;
    pid_t pid;
};

int main()
{
    int shmid;
    pid_t child_pid;
    struct shared_data *shared;

    shmid = shmget(IPC_PRIVATE, sizeof(struct shared_data), IPC_CREAT | 0600);
    if (shmid == -1) {
        perror("shmget");
        exit(1);
    }

    shared = (struct shared_data *)shmat(shmid, NULL, 0);
    if (shared == (void *)-1) {
        perror("shmat");
        shmctl(shmid, IPC_RMID, NULL);
        exit(1);
    }

    shared->state = 0;
    shared->pid = 0;

    child_pid = fork();
    if (child_pid < 0) {
        perror("fork");
        shmdt(shared);
        shmctl(shmid, IPC_RMID, NULL);
        exit(1);
    }

    if (child_pid == 0) {
        shared->pid = getpid();
        shared->state = 1;

        while (shared->state != 2) {
            usleep(10000);
        }

        printf("Child reads parent pid from shared memory: %d\n", shared->pid);

        shmdt(shared);
        exit(0);
    }

    while (shared->state != 1) {
        usleep(10000);
    }

    printf("Parent reads child pid from shared memory: %d\n", shared->pid);

    shared->pid = getpid();
    shared->state = 2;

    wait(NULL);

    shmdt(shared);
    shmctl(shmid, IPC_RMID, NULL);

    return 0;
}
```

这份程序完成讲义最后一题：子进程先写自己的进程号，父进程读出；父进程再写自己的进程号，子进程读出；最后由父进程注销共享内存。
